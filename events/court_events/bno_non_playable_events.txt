namespace = bno_non_playable

#landless bbc will steal playable gold
bno_non_playable.0001 = {
    hidden = yes
    trigger = {
        is_playable_character = no
        liege ?= {
            is_playable_character = yes

            OR = {
                is_bbc = no
                is_slut = no
            }
            gold > major_gold_value
            any_councillor = {
                has_council_position = councillor_steward
            }
        }
    }
    immediate = {
        liege = {
            random_councillor = {
                limit = {
                    has_council_position = councillor_steward
                }
                save_scope_as = cp_steward
            }
            send_interface_message = {
                title = bno_non_playable.0001.title
                desc = bno_non_playable.0001.desc
                remove_short_term_gold = minor_gold_value
            }
        }
        hidden_effect = {
            add_gold = minor_gold_value
        }
    }
}
