#hidden repeating event to pump up the var
bno_tincture_of_genesis.1111 = {
    hidden = yes
    immediate = {
        if = {
            limit = {
                var:bno_becoming_dark <= 4
            }
            change_variable = {
                name = bno_becoming_dark
                add = 0.006
            }
            trigger_event = {
                id = bno_tincture_of_genesis.1111
                days = 2
                delayed = yes
            }
        }
        else = {
            set_variable = {
                name = bno_becoming_dark
                value = 4
            }
        }
    }
}
