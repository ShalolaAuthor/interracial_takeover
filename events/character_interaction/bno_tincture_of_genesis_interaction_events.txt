#hidden repeating event to pump up the var
bno_tincture_of_genesis.1111 = {
    hidden = yes

    trigger = {
        scope:actor = { is_alive = yes }
        scope:recipient = { is_alive = yes }
    }

    on_trigger_fail = {
        remove_character_modifier = bno_tincture_of_genesis
        every_spouse = {
            limit = {
                has_character_modifier = bno_nourishers_fatigue
            }
            remove_character_modifier = bno_nourishers_fatigue
        }
    }

    immediate = {
        if = {
            limit = {
                has_trait = bno_genesis_transformation
                has_character_modifier = bno_tincture_of_genesis
            }
            scope:actor = {
                bno_have_sex_with_effects = {
                    PARTNER = scope:recipient
                }
            }
            trigger_event = {
                id = bno_tincture_of_genesis.1111
                days = 2
                delayed = yes
            }
            if = {
                limit = {
                    bno_genesis_transformation >= 1
                    NOT = { has_character_flag = bno_tincture_of_genesis.2002 }
                }
                trigger_event = bno_tincture_of_genesis.2002
            }
            else_if = {
                limit = {
                    bno_genesis_transformation >= 2
                    NOT = { has_character_flag = bno_tincture_of_genesis.4002 }
                }
                trigger_event = bno_tincture_of_genesis.4002
            }
            else_if = {
                limit = {
                    bno_genesis_transformation >= 3
                    NOT = { has_character_flag = bno_tincture_of_genesis.5002 }
                }
                trigger_event = bno_tincture_of_genesis.5002
            }
            else_if = {
                limit = {
                    bno_genesis_transformation >= 4
                    NOT = { has_character_flag = bno_tincture_of_genesis.6002 }
                }
                trigger_event = bno_tincture_of_genesis.6002
            }
        }
    }
}
