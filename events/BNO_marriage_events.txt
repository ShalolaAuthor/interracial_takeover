namespace = bno_marriage

bno_marriage.0001 = {
	hidden = yes
	
	cooldown = { months = { 12 18 } }
	
	trigger = {
		is_ai_blackbull = yes
		is_married = no
		is_ruler = no
		does_something_exist = {
			TRIGGER_KEY = is_lover_unmarried_ai_pure_white_girl_not_wearing_chastity_piercing
		}
	}
	
	immediate = {
		get_something = {
			TRIGGER_KEY = is_lover_unmarried_ai_pure_white_girl_not_wearing_chastity_piercing
			SCOPE_NAME = target1
		}
		random_list = {#TODO: add liege message
			1 = {
				bno_have_sex_with_effects = {
					PARTNER = scope:target1
				}
				become_lover_with_target = {
					TARGET = scope:target1
					LOCATION = location
				}
				save_scope_as = bbc
				scope:target1 ?= {
					bno_marriage_effect = {
						TARGET = scope:bbc
					}
				}
			}
			4 = {}
		}
	}
}

bno_marriage.0006 = {	#used to make sure bbc actually have children. Might be worthless/overburdening to have too many characters having kids, so not currently added to any event pulses
	hidden = yes
	
	cooldown = { months = { 12 18 } }

	trigger = {
		is_ai_blackbull = yes
		is_married = yes
		any_spouse = {
			is_a_girl = yes
		}
		bno_num_mixed_children_plus_dead <= 3
	}

	immediate = {
		every_spouse = {
			limit = {
				is_a_girl = yes
			}
			save_scope_as = wife_temp
			scope:wife_temp ?= {
				bno_have_sex_with_effects = {
					PARTNER = root
				}
			}
		}
	}
}

					}
				}
			}

				dynasty ?= {

					}
				}
			}
		}
	}
