# assigning the fluctuation of traits and their development. "becoming sluttier"

become_more_cuck_effect = {
	if = {
		limit = {
			NOT = {
				has_trait = bno_whiteboi
			}
			is_bbc = no
		}
		add_trait = bno_cuckish
	}
	else_if = {
		limit = {
			is_bbc = no
		}
		change_trait_rank = {
			trait = bno_whiteboi
			rank = 1
		}
	}
}

bno_castrated_effect = {
	switch = {
		trigger = has_trait
		lustful = { remove_trait = lustful }
		fecund = { remove_trait = fecund }
		bno_chastity = { remove_trait = bno_chastity }
	}
	add_prestige = -500
	add_trait = chaste
	add_trait = celibate
	add_trait = bno_castrated
}

bno_caged_effect = {

		trigger = has_trait
