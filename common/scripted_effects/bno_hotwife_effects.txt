bno_grateful_opinion = {
    add_opinion = {
        target = root
        modifier = grateful_opinion
        opinion = $VALUE$
    }
}

bno_hate_opinion = {
    add_opinion = {
        target = root
        modifier = hate_opinion
        opinion = $VALUE$
    }
}

bno_add_random_army_small_effect = {
    random_list = {
        1 = {
            if = {
                limit = { exists = var:bno_hotwife_army }
                spawn_army = {
                    men_at_arms = {
                        type = bno_ebony_elites
                        men = { 1 5 }
                    }
                    location = location
                    army = var:bno_hotwife_army
                    name = bno_hotwife_army_name
                }
            }
            else = {
                spawn_army = {
                    men_at_arms = {
                        type = bno_ebony_elites
                        men = { 1 5 }
                    }
                    location = location
                    name = bno_hotwife_army_name
                    save_scope_as = bno_hotwife_army
                }
                set_variable = {
                    name = bno_hotwife_army
                    value = scope:bno_hotwife_army
                }
            }
        }
        4 = {
            if = {
                limit = { exists = var:bno_hotwife_army }
                spawn_army = {

                        type = bno_african_warriors

                    name = bno_hotwife_army_name
                }
            }
            else = {
                spawn_army = {
                    men_at_arms = {

                    }
                    location = location
                    name = bno_hotwife_army_name

bno_have_sex_with_effects_multiple_times = {

        }
    }
}
