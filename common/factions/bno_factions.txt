bnwo_faction = {
    casus_belli = bno_civil_unrest

    leaders_allowed_to_leave = no
    player_can_join = no

    sort_order = 999

    name = bnwo_faction.name

    short_effect_desc = bnwo_faction.short_effect_desc

    discontent_progress = {
        base = 0
        common_discontent_progress_modifier = yes
    }

    power_threshold = {
        base = 80

        modifier = {
            add = 20
            faction_target = {
                NOT = { culture = culture:bnoc }
            }
            desc = bnwo_faction.power.bnoc
        }

        modifier = {
            add = 20
            faction_target = {
                NOT = { faith = faith:bbcs }
            }
            desc = bnwo_faction.power.bbcs
        }
    }

    requires_character = no

    #Faction existence is valid
    is_valid = {
        trigger_if = {
            limit = {
                exists = special_character
            }
            special_character = {
                is_alive = yes
            }
        }
    }

    #character valid check
    is_character_valid = {
        is_ai = yes
        has_trait = bno_bbc
        NOT = { has_character_flag = joining_faction_block }
    }   

    #county valid check
    is_county_valid = {
        custom_tooltip = {
            text = bnwo_faction.not_african_or_bbc_church
            OR = {
                NOT = { culture = culture:bnoc }
                NOT = { faith = faith:bbcs }
            }
        }

        #Preventing game over
        save_temporary_scope_as = this_county
        trigger_if = {
            limit = {
                holder = {
                    is_ai = no
                }
            }
            NOT = { this = holder.capital_county }
        }
    }

    can_character_join = {
        culture = culture:bnoc
        faith = faith:bbcs
    }

    can_county_join = {
        custom_tooltip = {
            text = bnwo_faction.not_african_or_bbc_church
            OR = {
                NOT = { culture = culture:bnoc }
                NOT = { faith = faith:bbcs }
            }
        }
    }

    can_county_create = {
        county_opinion < 0

        custom_tooltip = {
            text = bnwo_faction.not_african_or_bbc_church
            OR = {
                NOT = { culture = culture:bnoc }
                NOT = { faith = faith:bbcs }
            }
        }
    }

    
    
}
