is_bno_actor_holder = {
	scope:holder ?= {
		OR = {
			is_bno_actor = yes
			liege ?= {
				is_bno_actor = yes
			}
			top_liege ?= {
				is_bno_actor = yes
			}
		}
	}
}

bno_church_trigger_key = {
	bno_has_embrace_african_5 = yes
	piety_level >= 3
}

is_bno_actor_has_relevant_dynasty_perks = {
	scope:holder ?= {
		OR = {
			$TRIGGER_KEY$ = yes
			liege ?= {
				$TRIGGER_KEY$ = yes
			}
			top_liege ?= {
				$TRIGGER_KEY$ = yes
			}
		}
	}
}
