bno_is_adult_bbc_or_whiteboi = {
    is_adult = yes
	NOR = {
        has_trait = bno_castrated
        has_trait = bno_castrated_futa
    }
	OR = {
		is_bbc = yes
		is_whiteboi = yes
	}
}

}

	is_at_war = no
	bno_illness_list = yes

				type = bno_bbc_priestess_court_position
			}
		}

		is_a_girl = yes

				type = bno_bbc_priest_court_position
			}
		}
	}
}
