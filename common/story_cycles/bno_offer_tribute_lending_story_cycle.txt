story_bno_offer_tribute_lending = {
    #for returning tribute once the lending period is done
    #available variable: bno_offer_tribute_target, bno_offer_tribute_owner
    on_owner_death = {
        bno_remove_from_tributes_and_patrons_list = yes
        if = {
            limit = { scope:owner = { is_alive = yes } }
            scope:owner = {
                send_interface_message = {
                    type = bno_blacked_events
                    title = bno_offer_tribute_lending_died_title
                    desc = bno_offer_tribute_lending_died_desc
                    left_icon = scope:tribute
                    scope:target = {
                        pay_short_term_gold = {
                            gold = minor_gold_value
                            target = scope:owner
                        }
                    }
                }
            }
        }
        end_story = yes
    }
    effect_group = {
        days = { 15 45 }
        trigger = {
            var:bno_offer_tribute_target = {
                is_alive = no
            }
        }
        triggered_effect = {
            trigger = { always = yes }
            effect = {
                bno_return_tribute_to_parent = yes
            }
        }
    }
    effect_group = {
        years = 5
        triggered_effect = {
            trigger = { always = yes }
            effect = {
                bno_return_tribute_to_parent = yes
            }
        }
    }
}
