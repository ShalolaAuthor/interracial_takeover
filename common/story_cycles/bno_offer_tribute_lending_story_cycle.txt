story_bno_offer_tribute_lending = {
    on_owner_death = {
        bno_remove_from_tributes_and_patrons_list = yes
        if = {
            limit = { scope:owner = { is_alive = yes } }
            scope:owner = {
                send_interface_message = {
                    type = bno_blacked_events
                    title = bno_offer_tribute_lending_died_title
                    desc = bno_offer_tribute_lending_died_desc
                    left_icon = scope:tribute
                    scope:target = {
                        pay_short_term_gold = {
                            gold = minor_gold_value
                            target = scope:owner
                        }
                    }
                }
            }
        }
        end_story = yes
    }
    effect_group = {
        years = 5
        triggered_effect = {
            trigger = { always = yes }
            effect = {
                bno_remove_from_tributes_and_patrons_list = yes
                story_owner = {
                    remove_relation_bno_slave = scope:target
                    scope:target = {
                        remove_courtier_or_guest = scope:tribute
                    }
                    if = {
                        limit = {
                            any_parent = {
                                is_alive = yes
                                OR = {
                                    is_ruler = yes
                                    any_spouse = {
                                        is_ruler = yes
                                    }
                                }
                            }
                        }
                        random_parent = {
                            limit = {
                                is_alive = yes
                                OR = {
                                    is_ruler = yes
                                    any_spouse = {
                                        is_ruler = yes
                                    }
                                }
                            }
                            save_scope_as = parent
                        }
                        start_travel_plan = {
                            destination = scope:parent.location
                            travel_leader = scope:tribute
                            travel_with_domicile = yes
                            players_use_planner = no
                            return_trip = no
                            can_cancel_planning = no
                            on_start_on_action = bno_offer_tribute_after_lending_start
                            on_arrival_on_action = bno_offer_tribute_after_lending_arrival
                            on_arrival_destinations = last
                        }
                    }
                }
                end_story = yes
            }
        }
    }
}
