	icon = bno_bbc_cum_addiction

	desc = bno_bbc_cum_addiction_interaction.desc

	interface_priority = 999
	scheme = bno_bbc_cum_addiction
	common_interaction = yes

	send_name = bno_bbc_cum_addiction_interaction.send_name
	ignores_pending_interaction_block = yes
	auto_accept = yes
	is_shown = {
        scope:actor = {
            is_adult = yes
			is_bbc = yes

        scope:recipient = {
			is_adult = yes
			is_a_girl = yes
			is_african = no
			NOT = { has_trait = bno_cum_addiction_4 }

				target_character = scope:recipient
			}
		}
	}

	on_accept = {
		scope:actor = {
			send_interface_toast = {
				type = bno_blacked_events
				title = bno_bbc_cum_addiction_interaction.title
				desc = bno_bbc_cum_addiction_interaction.tooltip

				right_icon = scope:recipient

				start_scheme = {
					type = bno_bbc_cum_addiction
					target_character = scope:recipient
				}

				seduce_interaction_stress_effect = yes
			}
		}
	}
	
	ai_targets = {
		ai_recipients = family
		max = 2
	}
	ai_targets = {
		ai_recipients = guests
		ai_recipients = courtiers
		max = 10
	}
	
	ai_target_quick_trigger = {

		attracted_to_owner = yes

	}

	ai_potential = {

		is_bbc = yes

	ai_will_do = {
		base = 0
		modifier = {
			add = 1000

			is_adult = yes
		}		
	}
