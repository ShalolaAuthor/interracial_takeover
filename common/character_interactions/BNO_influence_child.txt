bno_influence_child = {
	category = interaction_category_friendly
	common_interaction = yes
	interface_priority = 50
	icon = bno_curious
	ai_maybe = yes
	ai_min_reply_days = 4
	ai_max_reply_days = 9
	can_send_despite_rejection = yes
	popup_on_receive = yes
	pause_on_receive = yes
	
	desc = bno_influence_child_desc
	
	is_shown = { #any ai ward/hostage
		scope:recipient = {
			is_ai = yes
			is_adult = no
			is_a_girl = yes
			OR = {
				scope:actor = { has_relation_ward = scope:recipient }
				is_hostage_of = scope:actor
			}
			NOT = { has_trait = bno_whitegirl }
		}
		
		scope:actor = {
			is_bno_actor = yes
		}
	}
	
	cooldown_against_recipient = { years = 2 }
	
	on_accept = {
		bno_influence_child_success = yes
	}
	
	on_decline = {
		bno_influence_child_failure = yes
	}
	
	auto_accept = no
	
	ai_accept = {
		base = 0 # Try to make it 0 for most interactions
		bno_influence_child_modifiers = yes
	}
}
