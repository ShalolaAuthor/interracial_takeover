# character interaction for sneaking a BBC into a foriegn court

sneak_bbc = {
	category = interaction_category_vassal
	common_interaction = yes
	force_notification = yes
	greeting = negative
	desc = sneak_bbc_desc
	
	populate_actor_list = {
		scope:actor = {
			every_courtier = {
				add_to_list = characters
			}
		}
	}
	
	is_shown = {
		NOT = { scope:recipient = scope:actor }
		scope:recipient = { is_ruler = yes }
		scope:actor = {
			OR = {
				has_trait = bno_bbc
				has_trait = bno_qos
			}
		}
	}
	
	can_be_picked = {
		NOT = { scope:recipient = root }
		has_trait = bno_bbc
	}
	
	on_accept = {
		scope:actor = {
			send_interface_toast = {
				title = sneak_bbc
				right_icon = scope:recipient
				left_icon = scope:secondary_actor
				
				scope:recipient = {
					add_courtier = scope:secondary_actor
				}
			}
		}
	}
	
	ai_accept = {
		base = -30
		opinion_modifier = { # Opinion Factor
			who = scope:recipient
			opinion_target = scope:actor
			multiplier = 1.0
			desc = AI_OPINION_REASON
		}
		modifier = {
			add = 5
			desc = BBC_cg_spy_reason
			trigger = {
				scope:recipient = {
					any_courtier_or_guest = {
						has_trait = bno_whitegirl
					}
				}
			}
		}
		
		modifier = {
			add = {value = 10}
			desc = BBC_family_spy_reason
			trigger = {
				scope:recipient = {
					any_close_family_member = {
						has_trait = bno_whitegirl
					}
				}
			}
		}
		
		modifier = {
			add = {value = 10}
			desc = BBC_curious_reason
			trigger = {
				scope:recipient = {
					has_trait = bno_curious
				}
			}
		}
		
		modifier = {
			add = {value = 25}
			desc = BBC_fucked_reason
			trigger = {
				scope:recipient = {
					OR = {
						has_trait = bno_snowbunny
						has_trait = bno_blacked
						has_trait = bno_mindless_bbc_slut
					}
				}
			}
		}
		
		modifier = {
			add = {value = 50}
			desc = BBC_slut_reason
			trigger = {
				scope:recipient = {
					has_trait = bno_mindless_bbc_slut
				}
			}
		}
		
		modifier = {
			add = {value = 50}
			desc = BBC_welcome_reason
			trigger = {
				scope:recipient = {
					has_trait = bno_bbc
				}
			}
		}
	}
}
