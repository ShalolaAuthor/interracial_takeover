		is_ai = no
		is_ruler = yes
        is_elligible_partner = yes
	}

		# Player ruler
		bno_random_random_events_lists = yes

		# Your courtiers and guests
		every_courtier_or_guest = {
            limit = {
                is_elligible_partner = yes
            }
			bno_random_random_events_lists = yes
		}

		# Neighbor rulers and their courts
		capital_county = {
			every_neighboring_county = {
				holder = {
					limit = { 
                        is_ruler = yes
                        is_elligible_partner = yes 
                    }
					bno_random_random_events_lists = yes
					
					every_courtier_or_guest = {
                        limit = {
                            is_elligible_partner = yes
                        }
						bno_random_random_events_lists = yes
					}
				}
			}
		}
	}
}
