on_game_start = {
    on_actions = {
        bno_on_game_start
    }
}

bno_on_game_start = {
    effect = {
        every_religion_global = {
            every_faith = {
                limit = {
                    NOT = {
                        religion_tag = bnonr_religion
                    }
                }
                add_doctrine = tenet_bno_hate
            }
        }
        every_culture_global = {
            limit = {
                trigger_if = {
                    limit = { has_game_rule = bno_african_culture_no }
                    bno_is_culture_african = no
                }
                trigger_else = {
                    NOT = { has_same_culture_heritage = culture:bnoc }
                }
            }
            add_culture_tradition = tradition_bno_african_hate
        }
    }
}
