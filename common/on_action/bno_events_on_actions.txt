	trigger = {
		is_a_girl = yes
		is_adult = yes
	}
	first_valid_on_action = {
		bno_slut_pulse
		bno_random_pulse
		bno_embrace_african_pulse
		bno_on_action_flag_reset_pulse
	}
}

	effect = {
		remove_character_flag = bno_recently_get_bull_events
		remove_character_flag = bno_recently_get_random_events
		remove_character_flag = bno_recently_get_embrace_african_events
		remove_character_flag = bno_recently_get_slut_events
		remove_character_flag = bno_recently_get_hotwife_events
		remove_character_flag = bno_recently_get_cuck_events
