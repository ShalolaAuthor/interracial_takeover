random_yearly_everyone_pulse = {
	on_actions = {
		bno_events_quarterly_pulse
		bno_events_half_year_pulse
		bno_events_each_year_pulse
	}
}

bno_events_pulse = {
	trigger = {
		has_game_rule = bno_mod_status_on
	}
	random_on_action = {
		1 = bno_events_slut_pulse
		1 = bno_events_bull_pulse
		1 = bno_events_cuck_pulse
	}
}

bno_events_slut_pulse = {
	trigger = {
		is_a_girl = yes
	}
	random_on_action = {
		1 = bno_slut_pulse
		1 = bno_random_pulse
		1 = bno_embrace_african_pulse
	}
}

bno_events_bull_pulse = {
	trigger = {
		is_bbc = yes
	}
	random_on_action = {
		1 = bno_bull_pulse
		1 = bno_random_pulse
		1 = bno_embrace_african_pulse
	}
}

bno_events_cuck_pulse = {
	trigger = {
		has_trait = bno_whiteboi
	}
	random_on_action = {
		1 = bno_cuck_pulse
		1 = bno_random_pulse
		1 = bno_embrace_african_pulse
	}
}

bno_events_quarterly_pulse = {
	trigger = {
		has_game_rule = bno_event_frequency_three_months
	}
	on_actions = {
		delay = { days = { 1 89 } }
		bno_events_pulse
		delay = { days = { 90 179 } }
		bno_events_pulse
		delay = { days = { 180 269 } }
		bno_events_pulse
		delay = { days = { 270 365 } }
		bno_events_pulse
	}
}
