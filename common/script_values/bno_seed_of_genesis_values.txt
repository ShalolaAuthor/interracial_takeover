bno_genesis_transformation_xp = {
    switch = {
        trigger = has_game_rule
        bno_xp_gain_low = {
            value = { 0.1 0.5 }
        }
        bno_xp_gain_med = {
            value = { 0.5 1 }
        }
        bno_xp_gain_high = {
            value = { 1 1.5 }
        }
    }
}

bno_genesis_transformation_xp_amount = {
    value = "this.has_trait_xp(bno_genesis_transformation)"
}

bno_genesis_transformation = {
    value = bno_genesis_transformation_xp_amount
    multiply = 4
    divide = 100    
}

bno_genesis_transformation_value = {
	value = 0
	if = {
		limit = {
			is_body_modification_ok = yes
			has_trait = bno_genesis_transformation
		}
		add = bno_genesis_transformation
	}
}
